Codewars RGB To Hex Conversion kata: 

function rgb(r, g, b){
  const toRound = n => n < 0 ? 0 : n > 255 ? 255 : n;
  const toHex = n => {
    const output = n.toString(16).toUpperCase();
    return (output.length === 1 ? '0' : '') + output;
  }
  
  return [r, g, b]
    .map(toRound)
    .map(toHex)
    .join('');
}
